```python
from datetime import datetime
from enum import Enum

class Timeframe(str, Enum):
    H1 = "1H"
    H4 = "4H"
    DAILY = "D"

@dataclass
class SpotPrice:
    """Raw spot price from API."""
    timestamp: datetime
    price: float
    
@dataclass
class OHLC:
    """Aggregated candle data."""
    timestamp: datetime  # Candle open time
    open: float
    high: float
    low: float
    close: float
    timeframe: Timeframe
```

2. Create `repository.py` with SQLite repositories:

- `SpotPriceRepository`:
  - `save(spot: SpotPrice) -> None`: Insert spot price
  - `get_range(start: datetime, end: datetime) -> list[SpotPrice]`: Get prices in range
  - `get_latest(limit: int = 100) -> list[SpotPrice]`: Get recent prices

- `OHLCRepository`:
  - `save(candle: OHLC) -> None`: Insert/update candle (upsert by timestamp+timeframe)
  - `get_latest(timeframe: Timeframe, limit: int = 100) -> list[OHLC]`: Get recent candles
  - `get_range(timeframe: Timeframe, start: datetime, end: datetime) -> list[OHLC]`: Get candles in range

Both repositories:
- Auto-create tables on first use (CREATE TABLE IF NOT EXISTS)
- Use parameterized queries (prevent SQL injection)
- Database path from Settings or default to `gold_data.db`	

Table schemas:
```sql
CREATE TABLE IF NOT EXISTS spot_prices (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp TEXT NOT NULL,
    price REAL NOT NULL,
    UNIQUE(timestamp)
);

CREATE TABLE IF NOT EXISTS ohlc_candles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp TEXT NOT NULL,
    open REAL NOT NULL,
    high REAL NOT NULL,
    low REAL NOT NULL,
    close REAL NOT NULL,
    timeframe TEXT NOT NULL,
    UNIQUE(timestamp, timeframe)
);
```

```python
import asyncio
import logging

class DataScheduler:
    """Schedules periodic data fetches with error handling."""
    
    def __init__(
        self,
        fetcher: DataFetcher,
        spot_repo: SpotPriceRepository,
        aggregator: CandleAggregator,
        interval_seconds: int = 900,
    ):
        self.fetcher = fetcher
        self.spot_repo = spot_repo
        self.aggregator = aggregator
        self.interval = interval_seconds
        self._running = False
        self._task: asyncio.Task | None = None
        self.logger = logging.getLogger(__name__)
    
    async def start(self) -> None:
        """Start the scheduler."""
        self._running = True
        self._task = asyncio.create_task(self._run_loop())
    
    async def stop(self) -> None:
        """Stop the scheduler gracefully."""
        self._running = False
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass
    
    async def _run_loop(self) -> None:
        """Main scheduler loop."""
        while self._running:
            try:
                await self._fetch_and_store()
            except Exception as e:
                self.logger.error(f"Fetch failed: {e}")
                # Don't crash, continue to next interval
            
            await asyncio.sleep(self.interval)
```
