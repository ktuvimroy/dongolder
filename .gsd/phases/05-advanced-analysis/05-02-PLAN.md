   ```python
           Expected input columns:

               - rsi (RSI value)
               - macd, macd_signal (MACD values)
               - ema_21, ema_50 (EMA values)
   ```

3. Implement feature creation:
   - **Price-based features:**
     - `price_change_pct`: Daily percent change
     - `high_low_range`: (high - low) / close
     - `close_to_high`: (high - close) / high
   
   - **Lagged returns (for each period in LOOKBACK_PERIODS):**
     - `return_{N}d`: N-day return
     - `volatility_{N}d`: Rolling N-day volatility (std of returns)
   
   - **RSI features:**
     - `rsi`: Raw RSI value
     - `rsi_overbought`: Binary (1 if RSI > 70)
     - `rsi_oversold`: Binary (1 if RSI < 30)
   
   - **MACD features:**
     - `macd_histogram`: macd - macd_signal
     - `macd_crossover`: Sign change in histogram (diff of sign)
   
   - **EMA features:**
     - `price_vs_ema21`: (close - ema_21) / ema_21
     - `price_vs_ema50`: (close - ema_50) / ema_50
     - `ema_trend`: Binary (1 if ema_21 > ema_50)
   
   - **Bollinger features:**
     - `bb_position`: (close - bb_lower) / (bb_upper - bb_lower)
     - `bb_squeeze`: (bb_upper - bb_lower) / bb_middle

4. Create target column:
   ```python
   def create_target(self, df: pd.DataFrame, horizon: int = 1) -> pd.Series:
       """Create classification target: did price go up or down?
       
       Args:
           df: DataFrame with 'close' column
           horizon: Days ahead to look (default 1)
           
       Returns:
           Series with 1 (up), -1 (down), 0 (flat < 0.1% change)
       """
       future_return = df['close'].shift(-horizon) / df['close'] - 1
       target = pd.Series(0, index=df.index)
       target[future_return > 0.001] = 1   # Up more than 0.1%
       target[future_return < -0.001] = -1  # Down more than 0.1%
       return target
   ```

   ```python
   @dataclass
   class MLPrediction:
       direction: SignalDirection
       probability: float  # Confidence 0-1
       signal: SignalDirection  # Same as direction for consistency
       
       @classmethod
       def from_prediction(cls, pred: int, proba: float) -> "MLPrediction":
           """Create from sklearn prediction.
           
           Args:
               pred: -1 (bearish), 0 (neutral), 1 (bullish)
               proba: Max class probability
           """
           if pred == 1:
               direction = SignalDirection.BULLISH
           elif pred == -1:
               direction = SignalDirection.BEARISH
           else:
               direction = SignalDirection.NEUTRAL
           return cls(direction=direction, probability=proba, signal=direction)
   ```

   ```python
   from sklearn.ensemble import HistGradientBoostingClassifier
   from sklearn.model_selection import TimeSeriesSplit
   import joblib
   
   class PatternRecognizer:
       """ML-based pattern recognition using gradient boosting."""
       
       MODEL_PATH = "data/models/pattern_model.joblib"
       MIN_TRAINING_SAMPLES = 100
       
       def __init__(self, feature_engineer: FeatureEngineer, model_path: str | None = None):
           self.feature_engineer = feature_engineer
           self.model_path = model_path or self.MODEL_PATH
           self.model: HistGradientBoostingClassifier | None = None
           self._load_model_if_exists()
   ```

3. Implement training:
   ```python
   def train(self, df: pd.DataFrame, n_splits: int = 5) -> dict:
       """Train model on historical data.
       
       Args:
           df: DataFrame with OHLCV + indicator columns
           n_splits: Number of cross-validation splits
           
       Returns:
           Dict with training metrics (accuracy per fold, avg accuracy)
       """
       features = self.feature_engineer.create_features(df)
       target = self.feature_engineer.create_target(df)
       
       # Align features and target (drop NaN rows)
       aligned = features.join(target.rename('target')).dropna()
       X = aligned.drop('target', axis=1)
       y = aligned['target']
       
       if len(X) < self.MIN_TRAINING_SAMPLES:
           raise ValueError(f"Need {self.MIN_TRAINING_SAMPLES}+ samples, got {len(X)}")
       
       # TimeSeriesSplit for temporal validation
       tscv = TimeSeriesSplit(n_splits=n_splits)
       scores = []
       
       for train_idx, val_idx in tscv.split(X):
           X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
           y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]
           
           model = HistGradientBoostingClassifier(
               max_iter=100,
               max_depth=5,
               learning_rate=0.1,
               early_stopping=True,
               validation_fraction=0.1,
               random_state=42
           )
           model.fit(X_train, y_train)
           scores.append(model.score(X_val, y_val))
       
       # Final model on all data
       self.model = HistGradientBoostingClassifier(
           max_iter=100,
           max_depth=5,
           learning_rate=0.1,
           early_stopping=True,
           random_state=42
       )
       self.model.fit(X, y)
       self._save_model()
       
       return {"fold_scores": scores, "avg_accuracy": sum(scores) / len(scores)}
   ```

4. Implement prediction:
   ```python
   def predict(self, df: pd.DataFrame) -> MLPrediction | None:
       """Predict direction for latest row.
       
       Args:
           df: DataFrame with OHLCV + indicator columns (at least 20 rows for features)
           
       Returns:
           MLPrediction or None if model not trained
       """
       if self.model is None:
           return None
           
       features = self.feature_engineer.create_features(df)
       if features.empty:
           return None
           
       # Get latest row
       X = features.iloc[[-1]]
       pred = self.model.predict(X)[0]
       proba = self.model.predict_proba(X).max()
       
       return MLPrediction.from_prediction(int(pred), float(proba))
   ```

5. Implement model persistence:

   ```python
   def _save_model(self):
       """Save model to disk."""
       import os
       os.makedirs(os.path.dirname(self.model_path), exist_ok=True)
       joblib.dump(self.model, self.model_path)
   
   def _load_model_if_exists(self):
       """Load model from disk if exists."""
       import os
       if os.path.exists(self.model_path):
           self.model = joblib.load(self.model_path)
   ```
